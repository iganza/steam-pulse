ReviewRepository.find_review_stats(appid: int) -> dict
─────────────────────────────────────────────────────
SQL LOGIC:
  Query 1 (Timeline):
    SELECT DATE_TRUNC('week', posted_at) AS week,
           COUNT(*) AS total,
           COUNT(CASE WHEN voted_up THEN 1 END) AS positive,
           ROUND(COUNT(voted_up)::numeric / COUNT(*) * 100) AS pct_positive
    FROM reviews WHERE appid = %s AND posted_at IS NOT NULL
    GROUP BY 1 ORDER BY 1

  Query 2 (Playtime Buckets):
    SELECT
      CASE
        WHEN playtime_hours = 0     THEN '0h'
        WHEN playtime_hours < 2     THEN '<2h'
        WHEN playtime_hours < 10    THEN '2-10h'
        WHEN playtime_hours < 50    THEN '10-50h'
        WHEN playtime_hours < 200   THEN '50-200h'
        ELSE '200h+'
      END AS bucket,
      COUNT(*) AS reviews,
      ROUND(COUNT(voted_up)::numeric / COUNT(*) * 100) AS pct_positive
    FROM reviews WHERE appid = %s
    GROUP BY 1 ORDER BY MIN(playtime_hours)

  Velocity Calculation (in Python):
    reviews_per_day = total_reviews / max((today - first_week_date).days, 1)
    reviews_last_30_days = sum of reviews with week >= (today - 30 days)

RETURNS:
  {
    "timeline": [...],           # Array of week objects
    "playtime_buckets": [...],   # Array of bucket objects
    "review_velocity": {...}     # Dict with 2 metrics
  }

GameRepository.find_benchmarks(appid: int, genre: str, year: int, 
                               price: float|None, is_free: bool) -> dict
──────────────────────────────────────────────────────────────────────────
SQL LOGIC (Complex CTE):
  WITH cohort AS (
    SELECT g.appid, g.positive_pct, g.review_count
    FROM games g
    JOIN game_genres gg ON gg.appid = g.appid
    JOIN genres gn ON gg.genre_id = gn.id
    WHERE gn.name = %s                           -- genre
      AND EXTRACT(YEAR FROM g.release_date) = %s  -- year
      AND (
          (g.is_free = TRUE AND %s = TRUE)
          OR (g.price_usd BETWEEN %s * 0.5 AND %s * 2.0)  -- ±50-200% price range
      )
      AND g.review_count > 50
  ),
  ranked AS (
    SELECT appid,
           PERCENT_RANK() OVER (ORDER BY positive_pct) AS sentiment_rank,
           PERCENT_RANK() OVER (ORDER BY review_count) AS popularity_rank
    FROM cohort
  )
  SELECT r.sentiment_rank, r.popularity_rank,
         (SELECT COUNT(*) FROM cohort) AS cohort_size
  FROM ranked r WHERE r.appid = %s

EDGE CASES:
  - Game not in cohort (no genres, missing metadata): returns all NULLs + cohort_size=0
  - Cohort size < 2: PERCENT_RANK() may return NULL
  - No games in cohort: all fields NULL

RETURNS:
  {
    "sentiment_rank": 0.75 | None,
    "popularity_rank": 0.85 | None,
    "cohort_size": 145
  }

TagRepository.find_genres_for_game(appid: int) -> list[dict]
─────────────────────────────────────────────────────────────
SELECT g.id, g.name, g.slug
FROM genres g
JOIN game_genres gg ON gg.genre_id = g.id
WHERE gg.appid = %s

RETURNS: [{"id": int, "name": str, "slug": str}, ...]

TagRepository.find_tags_for_game(appid: int) -> list[dict]
──────────────────────────────────────────────────────────
SELECT t.id, t.name, t.slug, gt.votes
FROM tags t
JOIN game_tags gt ON gt.tag_id = t.id
WHERE gt.appid = %s
ORDER BY gt.votes DESC

RETURNS: [{"id": int, "name": str, "slug": str, "votes": int}, ...]
